Let's learn Angular Pipes

uppercase, lowercase, titlecase Pipes::
{{userName}}
{{userName | uppercase}}
{{userName | lowercase}}
{{userName | titlecase}}
Currency Pipe::
{{ salary }}
{{ salary | currency }}
{{ salary | currency : 'USD' }}
{{ salary | currency : 'INR' }}
{{ salary | currency : '€' }}
{{ salary | currency : '¥' }}
{{ salary | currency : 'USD' : 'code' }}
{{ salary | currency : 'USD' : 'symbol' }}
Date Pipe::
{{ dateObj }}
{{ dateObj | date }}
{{ dateObj | date :'short'}}
{{ dateObj | date :'medium'}}
{{ dateObj | date :'long'}}
{{ dateObj | date :'full'}}
==================
{{ dateObj | date : 'shortDate' }}
{{ dateObj | date : 'mediumDate' }}
{{ dateObj | date : 'longDate' }}
{{ dateObj | date : 'fullDate' }}
=================
{{ dateObj | date : 'shortTime' }}
{{ dateObj | date : 'mediumTime' }}
{{ dateObj | date : 'longTime' }}
{{ dateObj | date : 'fullTime' }}
==============
{{ dateObj | date : 'dd-MMMM-yyyy' }}
{{ dateObj | date : 'hh:mm:ss a' }}
{{ dateObj | date: "fullTime":"UTC" }}
{{ dateObj | date: "fullTime":"IST" }}
{{ dateObj | date: "EEEE" }}
number & percent pipe::
{{100000000 | number}}
{{100000000 | number:'1.0-0' : 'en-IN'}}
{{ 12.111222333 | number: "3.2-5" }}
{{ 12.1 | number: "3.2-5" }}
{{ 0.5 | number:'3.2-5'}}
{{ 0.5 | percent}}
{{ 0.05 | percent}}
{{ 0.25 | percent}}
json pipe::
{{user.name}} is {{user.age}} years old, he is from {{user.add}}
{{user}}
{{user | json}}
keyvalue pipe::
@for (entry of user | keyvalue; track $index) {
{{entry.key}} -- {{entry.value}}
}
slice pipe can be used with array and string
{{cars | slice : 2 : 4}}
{{'Angular is a Framework' | slice : 0 : 7}}
{{'Angular is a Framework' | slice : 0 : -4}}
Custom Pipe(remaining)::
Remaining: {{msg | remaining}}
Custom Pipe(ordinal)::
Ordinal value: {{num | ordinal}}
Custom Pipe(Roman)::
Roman value: {{num | roman}}
Custom Pipe(credit card)::
cardNumber: {{cardNumber | creditcard}}
Custom Pipe(Age)::
Age is : {{dob | age}}
